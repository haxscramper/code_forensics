207 template <typename T, typename... Rest>
208 inline void hash_combine(std::size_t& seed, const T& v, Rest... rest) {
209  std::hash<T> hasher;
210  seed ^= hasher(v) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
211  hash_combine(seed, rest...);
