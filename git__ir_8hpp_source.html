185 template <typename T, typename... Rest>
186 inline void hash_combine(std::size_t& seed, const T& v, Rest... rest) {
187  std::hash<T> hasher;
188  seed ^= hasher(v) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
189  hash_combine(seed, rest...);
